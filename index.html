git --version: kiem tra phien ban
git help: cac thanh cong cu co the su dung
git init: tao kho luu tru
git status: kiem tra trang thai
git add tên_file: đưa file từ working directory vào staging area
git add .: đưa tất cả file từ working directory vào staging area
git commit -m "nội dung": đưa staging area vào Repository
git log: kiểm tra thông tin lịch sử
git show commit-id: xem chi tiết commit
git diff: xem sự thay đổi của file sau khi chỉnh sửa
gitk: mở dashboard xem chi tiết hơn
git checkout tên_file: đưa về ban đầu
git reset tên_file: đưa file từ staging area về working directory
git reset --soft commit_id: chuyển file từ repository qua staging area
git reset --mixed commit_id : chuyển file từ Repository qua working directory
git branch: xem danh sách nhánh
git checkout -b tên_file: tạo nhánh và chuyển sang nhánh đó
git checkout ten_file: chuyển sang nhánh khác
git merge tên_file : ghép nhánh
git branch -D tên_nhánh: xóa nhánh
